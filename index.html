const canvas = document.getElementById('game'); const ctx = canvas.getContext('2d'); canvas.width = 400; canvas.height = 600; const images = {}; const assets = ['hero', 'pipe', 'bullet', 'boss', 'victory']; let loadedCount = 0; assets.forEach(name => { images[name] = new Image(); images[name].src = `assets/${name}.png`; images[name].onload = () => { loadedCount++; }; }); let bird, pipes, score, gameOver, pipeTimer; function reset() { bird = { x: 50, y: 300, w: 45, h: 45, v: 0, g: 0.45, jump: -8 }; pipes = []; score = 0; gameOver = false; pipeTimer = 0; } reset(); function flap() { if (gameOver) { reset(); return; } bird.v = bird.jump; } window.addEventListener('mousedown', flap); window.addEventListener('touchstart', (e) => { e.preventDefault(); flap(); }); function update() { if (gameOver) { draw(); requestAnimationFrame(update); return; } bird.v += bird.g; bird.y += bird.v; pipeTimer++; if (pipeTimer > 100) { let gap = 180; // Проход между трубами (сделал побольше) let minPipeHeight = 50; let pipeY = Math.random() * (canvas.height - gap - 2 * minPipeHeight) + minPipeHeight; // Добавляем ПАРУ труб: верхнюю и нижнюю pipes.push({ x: 400, y: pipeY, gap: gap, w: 60 }); pipeTimer = 0; } for (let i = pipes.length - 1; i >= 0; i--) { let p = pipes[i]; p.x -= 3.5; // ХИТБОКС (уменьшен на 15 пикселей с каждой стороны для честности) let bx = bird.x + 10; let by = bird.y + 10; let bw = bird.w - 20; let bh = bird.h - 20; if (bx + bw > p.x && bx < p.x + p.w) { // Столкновение с верхней трубой if (by < p.y) gameOver = true; // Столкновение с нижней трубой if (by + bh > p.y + p.gap) gameOver = true; } if (p.x + p.w < 0) { pipes.splice(i, 1); score++; } } if (bird.y + bird.h > canvas.height || bird.y < -20) gameOver = true; draw(); requestAnimationFrame(update); } function draw() { ctx.clearRect(0, 0, canvas.width, canvas.height); // Птица if (images.hero) ctx.drawImage(images.hero, bird.x, bird.y, bird.w, bird.h); // Трубы pipes.forEach(p => { if (images.pipe) { // Верхняя труба (перевернутая) ctx.save(); ctx.translate(p.x + p.w/2, p.y); ctx.scale(1, -1); ctx.drawImage(images.pipe, -p.w/2, 0, p.w, 600); ctx.restore(); // Нижняя труба ctx.drawImage(images.pipe, p.x, p.y + p.gap, p.w, 600); } }); ctx.fillStyle = "white"; ctx.strokeStyle = "black"; ctx.lineWidth = 4; ctx.font = "bold 40px Arial"; ctx.strokeText("Счет: " + score, 20, 50); ctx.fillText("Счет: " + score, 20, 50); if (gameOver) { ctx.fillStyle = "rgba(0,0,0,0.7)"; ctx.fillRect(0, 0, canvas.width, canvas.height); if (images.victory) ctx.drawImage(images.victory, 75, 100, 250, 250); ctx.fillStyle = "white"; ctx.textAlign = "center"; ctx.font = "bold 35px Arial"; ctx.fillText("ФИНИШ!", canvas.width/2, 400); ctx.font = "20px Arial"; ctx.fillText("Твой счет: " + score, canvas.width/2, 440); ctx.fillText("Нажми, чтобы повторить", canvas.width/2, 490); ctx.textAlign = "start"; } } update();
