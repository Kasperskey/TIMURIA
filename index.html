const canvas = document.getElementById('game'); const ctx = canvas.getContext('2d'); canvas.width = 400; canvas.height = 600; const images = {}; const assets = ['hero', 'pipe', 'bullet', 'boss', 'victory']; let loadedCount = 0; assets.forEach(name => { images[name] = new Image(); images[name].src = `assets/${name}.png`; images[name].onload = () => { loadedCount++; }; }); let bird, pipes, score, gameOver, pipeTimer; function reset() { // g: 0.25 - очень мягкое падение, jump: -6 - плавный прыжок bird = { x: 50, y: 200, w: 50, h: 50, v: 0, g: 0.25, jump: -6 }; pipes = []; score = 0; gameOver = false; pipeTimer = 0; } reset(); function flap() { if (gameOver) { reset(); return; } bird.v = bird.jump; } // Управление для ПК и Телефона window.addEventListener('mousedown', (e) => { if(e.button === 0) flap(); }); window.addEventListener('touchstart', (e) => { e.preventDefault(); flap(); }, {passive: false}); function update() { if (gameOver) { draw(); requestAnimationFrame(update); return; } bird.v += bird.g; bird.y += bird.v; // Трубы появляются раз в 120 кадров pipeTimer++; if (pipeTimer > 120) { // Труба будет иметь случайную высоту снизу let pipeH = Math.random() * 250 + 150; pipes.push({ x: 400, y: canvas.height - pipeH, w: 70, h: 600 }); pipeTimer = 0; } for (let i = pipes.length - 1; i >= 0; i--) { pipes[i].x -= 2.5; // Медленное движение труб let p = pipes[i]; // Точная проверка столкновения if (bird.x + bird.w - 15 > p.x && bird.x + 15 < p.x + p.w) { if (bird.y + bird.h - 15 > p.y) { gameOver = true; } } if (p.x + p.w < 0) { pipes.splice(i, 1); score++; } } // Границы экрана if (bird.y + bird.h > canvas.height || bird.y < -50) gameOver = true; draw(); requestAnimationFrame(update); } function draw() { ctx.clearRect(0, 0, canvas.width, canvas.height); // Птица if (images.hero) ctx.drawImage(images.hero, bird.x, bird.y, bird.w, bird.h); // Трубы (только нижние, самые надежные) pipes.forEach(p => { if (images.pipe) ctx.drawImage(images.pipe, p.x, p.y, p.w, p.h); }); // Текст счета ctx.fillStyle = "white"; ctx.font = "bold 35px Arial"; ctx.shadowColor = "black"; ctx.shadowBlur = 7; ctx.fillText("Счет: " + score, 20, 50); ctx.shadowBlur = 0; if (gameOver) { ctx.fillStyle = "rgba(0,0,0,0.6)"; ctx.fillRect(0, 0, canvas.width, canvas.height); if (images.victory) ctx.drawImage(images.victory, 75, 130, 250, 250); ctx.fillStyle = "white"; ctx.textAlign = "center"; ctx.font = "bold 30px Arial"; ctx.fillText("ПОПРОБУЙ ЕЩЕ!", canvas.width/2, 430); ctx.font = "20px Arial"; ctx.fillText("Твой результат: " + score, canvas.width/2, 470); ctx.textAlign = "start"; } } update();
