const canvas = document.getElementById('game'); const ctx = canvas.getContext('2d'); canvas.width = 400; canvas.height = 600; const images = {}; const assets = ['hero', 'pipe', 'bullet', 'boss', 'victory']; let loadedCount = 0; assets.forEach(name => { images[name] = new Image(); images[name].src = `assets/${name}.png`; images[name].onload = () => { loadedCount++; }; }); // Настройки let bird, pipes, score, gameOver, pipeTimer; function reset() { bird = { x: 50, y: 300, w: 50, h: 50, v: 0, g: 0.4, jump: -8 }; pipes = []; score = 0; gameOver = false; pipeTimer = 0; } reset(); // Инициализация при запуске function flap() { if (gameOver) { reset(); return; } bird.v = bird.jump; } window.addEventListener('mousedown', flap); window.addEventListener('touchstart', (e) => { e.preventDefault(); flap(); }); function update() { if (gameOver) { draw(); requestAnimationFrame(update); return; } bird.v += bird.g; bird.y += bird.v; // Создаем трубы каждые 120 кадров (~2 секунды) pipeTimer++; if (pipeTimer > 120) { let pipeHeight = Math.random() * 200 + 100; // Высота нижней трубы pipes.push({ x: 400, y: 600 - pipeHeight, w: 70, h: 500 }); pipeTimer = 0; } for (let i = pipes.length - 1; i >= 0; i--) { pipes[i].x -= 3; // Проверка столкновения let p = pipes[i]; if (bird.x + bird.w - 10 > p.x && bird.x + 10 < p.x + p.w) { if (bird.y + bird.h - 10 > p.y) { gameOver = true; } } // Удаление труб и счет if (p.x + p.w < 0) { pipes.splice(i, 1); score++; } } // Смерть от падения или вылета вверх if (bird.y + bird.h > canvas.height || bird.y < -50) { gameOver = true; } draw(); requestAnimationFrame(update); } function draw() { ctx.clearRect(0, 0, canvas.width, canvas.height); // Птица if (images.hero) ctx.drawImage(images.hero, bird.x, bird.y, bird.w, bird.h); // Трубы pipes.forEach(p => { if (images.pipe) ctx.drawImage(images.pipe, p.x, p.y, p.w, p.h); }); // Счет ctx.fillStyle = "white"; ctx.strokeStyle = "black"; ctx.lineWidth = 2; ctx.font = "bold 40px Arial"; ctx.strokeText("Счет: " + score, 20, 50); ctx.fillText("Счет: " + score, 20, 50); if (gameOver) { ctx.fillStyle = "rgba(0,0,0,0.6)"; ctx.fillRect(0, 0, canvas.width, canvas.height); if (images.victory) ctx.drawImage(images.victory, 75, 120, 250, 250); ctx.fillStyle = "white"; ctx.textAlign = "center"; ctx.font = "bold 30px Arial"; ctx.fillText("ИГРА ОКОНЧЕНА", canvas.width/2, 420); ctx.font = "20px Arial"; ctx.fillText("Нажми, чтобы начать заново", canvas.width/2, 470); ctx.textAlign = "start"; } } update();
